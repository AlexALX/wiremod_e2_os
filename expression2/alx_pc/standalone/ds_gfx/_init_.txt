@strict

#include "alx_pc/standalone/ds_gfx/_shared_"
#include "alx_pc/standalone/ds_gfx/text"
#include "alx_pc/standalone/ds_gfx/textqueue"

if (first() | duped()) {
    
    DsCurrentFont = table()
    DsText = table()
    
    DSWL = nowirelink()
    
    DSCache = table()
}

function dsInit(DS:wirelink, Width:number, Height:number) {
    DSWL = DS
    DsWidth = Width
    DsHeight = Height
    
    DSWL[1048569] = 2
    DSWL[1048574] = 0
    DSWL[1048575] = 1
    DSWL[1048573] = Width
    DSWL[1048572] = Height

    dsStopQueues()
}

function dsSetFont(Font:table)
{
    DsFont = Font["glyphs",table]
    DsFontWidth = Font["width",number]
    DsFontHeight = Font["height",number]
    DsFontSpacing = Font["spacing",number]
}
