function test_readbyte_normal()
{
    local Name = "READBYTE NORMAL"

    jpgTestData(
        toChar(0x12)
        + toChar(0x34)
        + toChar(0x56)
    )

    if (!assertCase(Name, 1, jpgReadByte(), 0x12)) {
        return
    }

    if (!assertCase(Name, 2, jpgReadByte(), 0x34)) {
        return
    }

    if (!assertCase(Name, 3, jpgReadByte(), 0x56)) {
        return
    }

    testPass(Name)
}
test_readbyte_normal()

function test_bit_single_byte()
{
    local Name = "BIT SINGLE BYTE"

    jpgTestData(
        toChar(0xB2)
    )

    local Expected = array(
        1,
        0,
        1,
        1,
        0,
        0,
        1,
        0
    )

    for (I = 1, Expected:count()) {

        if (!assertCase(
            Name,
            I,
            jpgBit(),
            Expected[I,number]
        )) {
            return
        }
    }

    testPass(Name)
}
test_bit_single_byte()

function test_bits()
{
    local Name = "BITS"

    #
    # One byte
    #

    jpgTestData(
        toChar(0xD2)
    )

    if (!assertCase(Name, 1, jpgBits(3), 6)) {
        return
    }

    if (!assertCase(Name, 2, jpgBits(5), 18)) {
        return
    }

    #
    # Across byte boundary
    #

    jpgTestData(
        toChar(0xF0)
        + toChar(0x0F)
    )

    #
    # 111100000000
    #

    if (!assertCase(Name, 3, jpgBits(12), 3840)) {
        return
    }

    #
    # Read remaining bits
    #

    if (!assertCase(Name, 4, jpgBits(4), 15)) {
        return
    }

    testPass(Name)
}
test_bits()
