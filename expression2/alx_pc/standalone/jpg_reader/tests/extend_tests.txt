@name Extend Tests

#include "alx_pc/standalone/jpg_reader/tests/_shared_"
#include "alx_pc/standalone/jpg_reader/decoder"

function test_extend(
    Name:string,
    Size:number,
    Input:array,
    Expected:array
)
{
    for (I = 1, Input:count()) {

        local Result = jpgExtend(
            Input[I,number],
            Size
        )

        if (!assertCase(
            Name,
            I,
            Result,
            Expected[I,number]
        )) {
            return
        }
    }

    testPass(Name)
}

test_extend(
    "EXTEND SIZE1",
    1,
    array(0,1),
    array(-1,1)
)

test_extend(
    "EXTEND SIZE2",
    2,
    array(0,1,2,3),
    array(-3,-2,2,3)
)

test_extend(
    "EXTEND SIZE3",
    3,
    array(0,1,2,3,4,5,6,7),
    array(-7,-6,-5,-4,4,5,6,7)
)

testResults()
