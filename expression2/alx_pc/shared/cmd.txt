@persist CMD:string CMDArg CMDA:table

if (first()|duped()) {
 
#include "alx_pc/shared/_main_"

function runCmd(Cmd:string,Delay,NoDebug) {
    CMD = Cmd
    timer("cmd",Delay)
}

function runCmd(Cmd:string,Delay) {
    if (DEBUG) { Delay = 1 }
    runCmd(Cmd,Delay,1)
}


function runCmd(Cmd:string,Delay,Arg:table,NoDebug) {
    CMD = Cmd
    CMDArg = 1
    CMDA = Arg
    timer("cmd",Delay)
}

function runCmd(Cmd:string,Delay,Arg:table) {
    if (DEBUG) { Delay = 1 }
    runCmd(Cmd,Delay,Arg,1)
}

function cmdClk() {
    if (CMD=="") { return }
    if (CMDArg) {
        local ID = CMDA:id()
        if (!CMD(CMDA)[number]) {
            if (CMDA:id()==ID) {
                CMDA = table()
                CMDArg = 0
            }
        }
    } else {
        CMD()
    }
}

function stopCmd() {
    stoptimer("cmd")
    CMD = ""
    CMDArg = 0
    CMDA = table()
}

#[ added in CPU/GPU etc
if (clk("cmd")) {
    cmdClk()   
}
]#

}
