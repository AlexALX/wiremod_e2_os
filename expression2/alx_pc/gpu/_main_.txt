function number in_array(Arr:array, Value) {
    for (I=1,Arr:count()) {
        if (Arr[I,number]==Value) { return 1 }
    }       
    return 0
}

function err() {
    dsSendDirect("err",BIOS,ERR_C)
    if (ERR_C!=-1) { exit() }
}

function egpClear() {
    EGP:egpClear()
    EGP:egpDrawTopLeft(1)
    LAST_I = 1 
    LAST_Y = 0 
    GpuData = table()
    if (EgpCleanHandler!="") {
        EgpCleanHandler()
    }
}

function dsClear() {
    DS[1048574] = 0
    DS[1048575] = 1
}

function reboot() {
    dsSendDirect("REBOOT",BIOS,0)   
}

function hddload(Num) {
    dsSendDirect("HDDInd",BIOS,Num)    
}
